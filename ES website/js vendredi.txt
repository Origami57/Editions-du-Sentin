<script>
window._echoScrollInit = true;
document.addEventListener('click', function(e) {
  var link = e.target.closest('a[href="#echo-ticker"]');
  if (!link) return;
  e.preventDefault();
  e.stopPropagation();
  var target = document.querySelector('#echo-ticker');
  if (!target) return;
  var start = window.scrollY;
  var end = target.getBoundingClientRect().top + window.scrollY - 200;
  var duration = 800;
  var startTime = null;
  function ease(t) { return t < 0.5 ? 2*t*t : -1+(4-2*t)*t; }
  function step(timestamp) {
    if (!startTime) startTime = timestamp;
    var progress = Math.min((timestamp - startTime) / duration, 1);
    window.scrollTo(0, start + (end - start) * ease(progress));
    if (progress < 1) requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}, true);
</script>

<script>
document.querySelectorAll('[data-depuis]').forEach(function(el) {
    if (!el) return;
    var annees = new Date().getFullYear() - parseInt(el.getAttribute('data-depuis'));
    if (el.tagName === 'SPAN' && !el.classList.length) {
        el.textContent = annees;
    } else {
        el.innerHTML = annees + '<sup>ans</sup>';
    }
});
	
	
</script>
